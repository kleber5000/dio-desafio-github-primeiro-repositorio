1 - cria repositorio
2 - abre o bash na pasta desejada
3 - git init

5 - git remote -v

7 - git push origin master

1 - git add . ou git add -A

1 - git pull origin master
2 - git add .
3 - git commit -m "MENSAGEM"
4 - git push origin master
